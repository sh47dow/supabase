### Create a record

This example how to create a new city:

<!-- prettier-ignore -->
```js {7-11}
import { createClient } from '@supabase/supabase-js'

const supabase = createClient('https://world.supabase.io', '1a2b-3c4d-5e6f-7g8h')

const getCities = async () => {
  try {
    let cities = await supabase
      .from('cities')
      .insert([
        { name: 'The Shire', country_id: 554 }
      ])
    return cities
  } catch (error) {
    console.log('Error: ', error)
  }
}
```

### Bulk create

This example how to create multiple cities with one request:

<!-- prettier-ignore -->
```js {7-12}
import { createClient } from '@supabase/supabase-js'

const supabase = createClient('https://world.supabase.io', '1a2b-3c4d-5e6f-7g8h')

const createCities = async () => {
  try {
    let cities = await supabase
      .from('cities')
      .insert([
        { name: 'The Shire', country_id: 554 },
        { name: 'Rohan', country_id: 554 },
      ])
    return cities
  } catch (error) {
    console.log('Error: ', error)
  }
}
```
